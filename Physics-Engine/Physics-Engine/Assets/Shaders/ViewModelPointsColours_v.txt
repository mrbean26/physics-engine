#version 330 core

layout(location = 0) in vec3 vertex;
layout(location = 1) in vec3 colour;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec3 vertexColor;

void main(){
    gl_Position = projection * view * model * vec4(vertex, 1.0f);
    vertexColor = vertexColor * vertex;
}
